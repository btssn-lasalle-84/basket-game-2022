488 int IHM::jouerUnJeton(QString numeroPanier, QString equipe)
489 {
490  qDebug() << Q_FUNC_INFO << numeroPanier << equipe;
491  int tir = numeroPanier.toInt() - 1; // numéro de colonne
492  // on recherche la ligne
493  for(int ligne = 0; ligne < plateau[tir].size(); ++ligne)
494  {
495  if(plateau[tir][ligne] == CouleurJeton::AUCUN)
496  {
497  if(equipe == "R")
498  plateau[tir][ligne] = CouleurJeton::ROUGE;
499  else if(equipe == "J")
500  plateau[tir][ligne] = CouleurJeton::JAUNE;
501  return ligne;
502  }
503  }
504 
505  return -1; // aucune position trouvé
506 }
